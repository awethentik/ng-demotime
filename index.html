Combines angularJS + hammerJS + ng-hammer + hotkeys to show all your taps on a touch screen

ngDemotime (ngHammer-showTouches v2)

Combines angularJS + hammerJS + ng-hammer

This is using Hammer.js (v2) and Angular Hammer with angularJS. If you need to use hammerJS v1 use this.

Installation

Add jquery-1.10.2.min.js (before angular) *temporary

<script src="js/vendor/jquery-1.10.2.min.js"></script>

Add hammer.js (before angular)

<script src="js/vendor/hammer.js"></script>

Add ng-hammer.js (after angular)

<script src="js/vendor/angular-hammer.js"></script>

Add hotkeys.min.js (after angular)

<script src="js/vendor/hotkeys.min.js"></script>

Add CSS (colors, size & opacity can be changed here)

.lastTouch {
  display: none;
  position: fixed;
  z-index: 9999;
  height: 24px;
  width: 24px;
  top: 0;
  left: 0;
  pointer-events: none;
  background: rgba(255, 0, 0, 0.8);
  border-radius: 20px;
  margin-top: -12px;
  margin-left: -12px;
}

.lastTouchType {
    position: absolute;
    z-index: 9998;
    height: 24px;
    width: 60px;
    bottom: 0;
    right: 0;
    pointer-events: none;
    background: #ececec;
}

Usage

Add directive as attribute

<div hm-tap="showTouch($event)" hm-press="showTouch($event)" hm-doubletap="showTouch($event)" show-touches></div>

Toggle on/off by using "hot key" (spacebar + d).

Directive

'use strict';

/* App Module */
var myApp = angular.module('myApp', ['hmTouchEvents', 'cfp.hotkeys']);
/* Directives */
myApp.directive('showTouches', ['hotkeys', function(hotkeys){
    return {
        restrict: 'A',
        link: function(scope, element) {
            element.append('&lt;div class="lastTouchType"&gt;&lt;/div&gt;');
            element.append('&lt;div class="lastTouch"&gt;&lt;/div&gt;');
            scope.showTouch = function(evt){
                if(scope.demoMode){
                    $('.lastTouchType').text(evt.type);
                    $('.lastTouch').hide();
                    $('.lastTouch').show().css({
                        'top': evt.center.y,
                        'left': evt.center.x
                    });
                    clearTimeout(timer);
                    timer = setTimeout(function () {
                        $('.lastTouch').fadeOut("fast");
                    }, 100);
                }
            };

            scope.demoMode = false;
            hotkeys.add({
                combo: 'space+d',
                description: 'Toggle Demo Mode',
                callback: function() {
                    scope.demoMode = !scope.demoMode;
                    $('.lastTouchType').toggle();
                }
            });
       }
    }
}]);
